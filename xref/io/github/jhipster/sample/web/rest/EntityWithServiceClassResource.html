20  /**
21   * REST controller for managing EntityWithServiceClass.

23  @RestController
24  @RequestMapping("/api")
25  public class EntityWithServiceClassResource {
26  
27      private final Logger log = LoggerFactory.getLogger(EntityWithServiceClassResource.class);
28  
29      private static final String ENTITY_NAME = "entityWithServiceClass";
30  
31      private final EntityWithServiceClassService entityWithServiceClassService;
32  
33      public EntityWithServiceClassResource(EntityWithServiceClassService entityWithServiceClassService) {
34          this.entityWithServiceClassService = entityWithServiceClassService;
35      }
36  
37      /**
38       * POST  /entity-with-service-classes : Create a new entityWithServiceClass.
39       *
40       * @param entityWithServiceClass the entityWithServiceClass to create
41       * @return the ResponseEntity with status 201 (Created) and with body the new entityWithServiceClass, or with status 400 (Bad Request) if the entityWithServiceClass has already an ID
42       * @throws URISyntaxException if the Location URI syntax is incorrect
43       */
44      @PostMapping("/entity-with-service-classes")
45      @Timed
46      public ResponseEntity<EntityWithServiceClass> createEntityWithServiceClass(@RequestBody EntityWithServiceClass entityWithServiceClass) throws URISyntaxException {
47          log.debug("REST request to save EntityWithServiceClass : {}", entityWithServiceClass);
48          if (entityWithServiceClass.getId() != null) {
49              throw new BadRequestAlertException("A new entityWithServiceClass cannot already have an ID", ENTITY_NAME, "idexists");
50          }
51          EntityWithServiceClass result = entityWithServiceClassService.save(entityWithServiceClass);
52          return ResponseEntity.created(new URI("/api/entity-with-service-classes/" + result.getId()))
53              .headers(HeaderUtil.createEntityCreationAlert(ENTITY_NAME, result.getId().toString()))
54              .body(result);
55      }
56  
57      /**
58       * PUT  /entity-with-service-classes : Updates an existing entityWithServiceClass.
59       *
60       * @param entityWithServiceClass the entityWithServiceClass to update
61       * @return the ResponseEntity with status 200 (OK) and with body the updated entityWithServiceClass,
62       * or with status 400 (Bad Request) if the entityWithServiceClass is not valid,
63       * or with status 500 (Internal Server Error) if the entityWithServiceClass couldn't be updated
64       * @throws URISyntaxException if the Location URI syntax is incorrect
65       */
66      @PutMapping("/entity-with-service-classes")
67      @Timed
68      public ResponseEntity<EntityWithServiceClass> updateEntityWithServiceClass(@RequestBody EntityWithServiceClass entityWithServiceClass) throws URISyntaxException {
69          log.debug("REST request to update EntityWithServiceClass : {}", entityWithServiceClass);
70          if (entityWithServiceClass.getId() == null) {
71              return createEntityWithServiceClass(entityWithServiceClass);
72          }
73          EntityWithServiceClass result = entityWithServiceClassService.save(entityWithServiceClass);
74          return ResponseEntity.ok()
75              .headers(HeaderUtil.createEntityUpdateAlert(ENTITY_NAME, entityWithServiceClass.getId().toString()))
76              .body(result);
77      }
78  
79      /**
80       * GET  /entity-with-service-classes : get all the entityWithServiceClasses.
81       *
82       * @return the ResponseEntity with status 200 (OK) and the list of entityWithServiceClasses in body
83       */
84      @GetMapping("/entity-with-service-classes")
85      @Timed
86      public List<EntityWithServiceClass> getAllEntityWithServiceClasses() {
87          log.debug("REST request to get all EntityWithServiceClasses");
88          return entityWithServiceClassService.findAll();
89          }
90  
91      /**
92       * GET  /entity-with-service-classes/:id : get the "id" entityWithServiceClass.
93       *
94       * @param id the id of the entityWithServiceClass to retrieve
95       * @return the ResponseEntity with status 200 (OK) and with body the entityWithServiceClass, or with status 404 (Not Found)
96       */
97      @GetMapping("/entity-with-service-classes/{id}")
98      @Timed
99      public ResponseEntity<EntityWithServiceClass> getEntityWithServiceClass(@PathVariable Long id) {
100         log.debug("REST request to get EntityWithServiceClass : {}", id);
101         EntityWithServiceClass entityWithServiceClass = entityWithServiceClassService.findOne(id);
102         return ResponseUtil.wrapOrNotFound(Optional.ofNullable(entityWithServiceClass));
103     }
104 
105     /**
106      * DELETE  /entity-with-service-classes/:id : delete the "id" entityWithServiceClass.
107      *
108      * @param id the id of the entityWithServiceClass to delete
109      * @return the ResponseEntity with status 200 (OK)
110      */
111     @DeleteMapping("/entity-with-service-classes/{id}")
112     @Timed
113     public ResponseEntity<Void> deleteEntityWithServiceClass(@PathVariable Long id) {
114         log.debug("REST request to delete EntityWithServiceClass : {}", id);
115         entityWithServiceClassService.delete(id);
116         return ResponseEntity.ok().headers(HeaderUtil.createEntityDeletionAlert(ENTITY_NAME, id.toString())).build();
117     }
