45  /**
46   * REST controller for managing the current user's account.
47   */
48  @RestController
49  @RequestMapping("/api")
50  public class AccountResource {
51  
52      private final Logger log = LoggerFactory.getLogger(AccountResource.class);
53  
54      private final UserRepository userRepository;
55  
56      private final UserService userService;
57  
58      private final MailService mailService;
59  
60      public AccountResource(UserRepository userRepository, UserService userService, MailService mailService) {
61  
62          this.userRepository = userRepository;
63          this.userService = userService;
64          this.mailService = mailService;
65      }
66  
67      /**
68       * POST  /register : register the user.
69       *
70       * @param managedUserVM the managed user View Model
71       * @throws InvalidPasswordException 400 (Bad Request) if the password is incorrect
72       * @throws EmailAlreadyUsedException 400 (Bad Request) if the email is already used
73       * @throws LoginAlreadyUsedException 400 (Bad Request) if the login is already used
74       */
75      @PostMapping("/register")
76      @Timed
77      @ResponseStatus(HttpStatus.CREATED)
78      public void registerAccount(@Valid @RequestBody ManagedUserVM managedUserVM) {
79          if (!checkPasswordLength(managedUserVM.getPassword())) {
80              throw new InvalidPasswordException();
81          }
82          userRepository.findOneByLogin(managedUserVM.getLogin().toLowerCase()).ifPresent(u -> {throw new LoginAlreadyUsedException();});
83          userRepository.findOneByEmailIgnoreCase(managedUserVM.getEmail()).ifPresent(u -> {throw new EmailAlreadyUsedException();});
84          User user = userService.registerUser(managedUserVM, managedUserVM.getPassword());
85          mailService.sendActivationEmail(user);
86      }
87  
88      /**
89       * GET  /activate : activate the registered user.
90       *
91       * @param key the activation key
92       * @throws RuntimeException 500 (Internal Server Error) if the user couldn't be activated
93       */
94      @GetMapping("/activate")
95      @Timed
96      public void activateAccount(@RequestParam(value = "key") String key) {
97          Optional<User> user = userService.activateRegistration(key);
98          if (!user.isPresent()) {
99              throw new InternalServerErrorException("No user was found for this reset key");
100         }
101     }
102 
103     /**
104      * GET  /authenticate : check if the user is authenticated, and return its login.
105      *
106      * @param request the HTTP request
107      * @return the login if the user is authenticated
108      */
109     @GetMapping("/authenticate")
110     @Timed
111     public String isAuthenticated(HttpServletRequest request) {
112         log.debug("REST request to check if the current user is authenticated");
113         return request.getRemoteUser();
114     }
115 
116     /**
117      * GET  /account : get the current user.
118      *
119      * @return the current user
120      * @throws RuntimeException 500 (Internal Server Error) if the user couldn't be returned
121      */
122     @GetMapping("/account")
123     @Timed
124     public UserDTO getAccount() {
125         return userService.getUserWithAuthorities()
126             .map(UserDTO::new)
127             .orElseThrow(() -> new InternalServerErrorException("User could not be found"));
128     }
129 
130     /**
131      * POST  /account : update the current user information.
132      *
133      * @param userDTO the current user information
134      * @throws EmailAlreadyUsedException 400 (Bad Request) if the email is already used
135      * @throws RuntimeException 500 (Internal Server Error) if the user login wasn't found
136      */
137     @PostMapping("/account")
138     @Timed
139     public void saveAccount(@Valid @RequestBody UserDTO userDTO) {
140         final String userLogin = SecurityUtils.getCurrentUserLogin().orElseThrow(() -> new InternalServerErrorException("Current user login not found"));
141         Optional<User> existingUser = userRepository.findOneByEmailIgnoreCase(userDTO.getEmail());
142         if (existingUser.isPresent() && (!existingUser.get().getLogin().equalsIgnoreCase(userLogin))) {
143             throw new EmailAlreadyUsedException();
144         }
145         Optional<User> user = userRepository.findOneByLogin(userLogin);
146         if (!user.isPresent()) {
147             throw new InternalServerErrorException("User could not be found");
148         }
149         userService.updateUser(userDTO.getFirstName(), userDTO.getLastName(), userDTO.getEmail(),
150             userDTO.getLangKey(), userDTO.getImageUrl());
151    }
152 
153     /**
154      * POST  /account/change-password : changes the current user's password
155      *
156      * @param password the new password
157      * @throws InvalidPasswordException 400 (Bad Request) if the new password is incorrect
158      */
159     @PostMapping(path = "/account/change-password")
160     @Timed
161     public void changePassword(@RequestBody String password) {
162         if (!checkPasswordLength(password)) {
163             throw new InvalidPasswordException();
164         }
165         userService.changePassword(password);
166    }
167 
168     /**
169      * POST   /account/reset-password/init : Send an email to reset the password of the user
170      *
171      * @param mail the mail of the user
172      * @throws EmailNotFoundException 400 (Bad Request) if the email address is not registered
173      */
174     @PostMapping(path = "/account/reset-password/init")
175     @Timed
176     public void requestPasswordReset(@RequestBody String mail) {
177        mailService.sendPasswordResetMail(
178            userService.requestPasswordReset(mail)
179                .orElseThrow(EmailNotFoundException::new)
180        );
181     }
182 
183     /**
184      * POST   /account/reset-password/finish : Finish to reset the password of the user
185      *
186      * @param keyAndPassword the generated key and the new password
187      * @throws InvalidPasswordException 400 (Bad Request) if the password is incorrect
188      * @throws RuntimeException 500 (Internal Server Error) if the password could not be reset
189      */
190     @PostMapping(path = "/account/reset-password/finish")
191     @Timed
192     public void finishPasswordReset(@RequestBody KeyAndPasswordVM keyAndPassword) {
193         if (!checkPasswordLength(keyAndPassword.getNewPassword())) {
194             throw new InvalidPasswordException();
195         }
196         Optional<User> user =
197             userService.completePasswordReset(keyAndPassword.getNewPassword(), keyAndPassword.getKey());
198 
199         if (!user.isPresent()) {
200             throw new InternalServerErrorException("No user was found for this reset key");
201         }
202     }
203 
204     private static boolean checkPasswordLength(String password) {
205         return !StringUtils.isEmpty(password) &&
206             password.length() >= ManagedUserVM.PASSWORD_MIN_LENGTH &&
207             password.length() <= ManagedUserVM.PASSWORD_MAX_LENGTH;
208     }
