46  /**
47   * Service for executing complex queries for BankAccount entities in the database.
48   * The main input is a {@link BankAccountCriteria} which get's converted to {@link Specifications},
49   * in a way that all the filters must apply.
50   * It returns a {@link List} of {@link BankAccountDTO} or a {@link Page} of {@link BankAccountDTO} which fulfills the criteria.
51   */
52  @Service
53  @Transactional(readOnly = true)
54  public class BankAccountQueryService extends QueryService<BankAccount> {
55  
56      private final Logger log = LoggerFactory.getLogger(BankAccountQueryService.class);
57  
58  
59      private final BankAccountRepository bankAccountRepository;
60  
61      private final BankAccountMapper bankAccountMapper;
62  
63      public BankAccountQueryService(BankAccountRepository bankAccountRepository, BankAccountMapper bankAccountMapper) {
64          this.bankAccountRepository = bankAccountRepository;
65          this.bankAccountMapper = bankAccountMapper;
66      }
67  
68      /**
69       * Return a {@link List} of {@link BankAccountDTO} which matches the criteria from the database
70       * @param criteria The object which holds all the filters, which the entities should match.
71       * @return the matching entities.
72       */
73      @Transactional(readOnly = true)
74      public List<BankAccountDTO> findByCriteria(BankAccountCriteria criteria) {
75          log.debug("find by criteria : {}", criteria);
76          final Specifications<BankAccount> specification = createSpecification(criteria);
77          return bankAccountMapper.toDto(bankAccountRepository.findAll(specification));
78      }
79  
80      /**
81       * Return a {@link Page} of {@link BankAccountDTO} which matches the criteria from the database
82       * @param criteria The object which holds all the filters, which the entities should match.
83       * @param page The page, which should be returned.
84       * @return the matching entities.
85       */
86      @Transactional(readOnly = true)
87      public Page<BankAccountDTO> findByCriteria(BankAccountCriteria criteria, Pageable page) {
88          log.debug("find by criteria : {}, page: {}", criteria, page);
89          final Specifications<BankAccount> specification = createSpecification(criteria);
90          final Page<BankAccount> result = bankAccountRepository.findAll(specification, page);
91          return result.map(bankAccountMapper::toDto);
92      }
93  
94      /**
95       * Function to convert BankAccountCriteria to a {@link Specifications}
96       */
97      private Specifications<BankAccount> createSpecification(BankAccountCriteria criteria) {
98          Specifications<BankAccount> specification = Specifications.where(null);
99          if (criteria != null) {
100             if (criteria.getId() != null) {
101                 specification = specification.and(buildSpecification(criteria.getId(), BankAccount_.id));
102             }
103             if (criteria.getName() != null) {
104                 specification = specification.and(buildStringSpecification(criteria.getName(), BankAccount_.name));
105             }
106             if (criteria.getBankNumber() != null) {
107                 specification = specification.and(buildRangeSpecification(criteria.getBankNumber(), BankAccount_.bankNumber));
108             }
109             if (criteria.getAgencyNumber() != null) {
110                 specification = specification.and(buildRangeSpecification(criteria.getAgencyNumber(), BankAccount_.agencyNumber));
111             }
112             if (criteria.getLastOperationDuration() != null) {
113                 specification = specification.and(buildRangeSpecification(criteria.getLastOperationDuration(), BankAccount_.lastOperationDuration));
114             }
115             if (criteria.getMeanOperationDuration() != null) {
116                 specification = specification.and(buildRangeSpecification(criteria.getMeanOperationDuration(), BankAccount_.meanOperationDuration));
117             }
118             if (criteria.getBalance() != null) {
119                 specification = specification.and(buildRangeSpecification(criteria.getBalance(), BankAccount_.balance));
120             }
121             if (criteria.getOpeningDay() != null) {
122                 specification = specification.and(buildRangeSpecification(criteria.getOpeningDay(), BankAccount_.openingDay));
123             }
124             if (criteria.getLastOperationDate() != null) {
125                 specification = specification.and(buildRangeSpecification(criteria.getLastOperationDate(), BankAccount_.lastOperationDate));
126             }
127             if (criteria.getActive() != null) {
128                 specification = specification.and(buildSpecification(criteria.getActive(), BankAccount_.active));
129             }
130             if (criteria.getAccountType() != null) {
131                 specification = specification.and(buildSpecification(criteria.getAccountType(), BankAccount_.accountType));
132             }
133             if (criteria.getUserId() != null) {
134                 specification = specification.and(buildReferringEntitySpecification(criteria.getUserId(), BankAccount_.user, User_.id));
135             }
136             if (criteria.getOperationId() != null) {
137                 specification = specification.and(buildReferringEntitySpecification(criteria.getOperationId(), BankAccount_.operations, Operation_.id));
138             }
139         }
140         return specification;
141     }
