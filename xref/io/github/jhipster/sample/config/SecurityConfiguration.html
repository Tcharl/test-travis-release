23  import io.github.jhipster.sample.security.*;
24  import io.github.jhipster.sample.security.jwt.*;
25  
26  import org.springframework.beans.factory.BeanInitializationException;
27  import org.springframework.context.annotation.Bean;
28  import org.springframework.context.annotation.Configuration;
29  import org.springframework.context.annotation.Import;
30  import org.springframework.http.HttpMethod;
31  import org.springframework.security.config.annotation.authentication.builders.AuthenticationManagerBuilder;
32  import org.springframework.security.config.annotation.method.configuration.EnableGlobalMethodSecurity;
33  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
34  import org.springframework.security.config.annotation.web.builders.WebSecurity;
35  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
36  import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
37  import org.springframework.security.config.http.SessionCreationPolicy;
38  import org.springframework.security.core.userdetails.UserDetailsService;
39  import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
40  import org.springframework.security.crypto.password.PasswordEncoder;
41  import org.springframework.security.data.repository.query.SecurityEvaluationContextExtension;
42  import org.springframework.security.web.authentication.UsernamePasswordAuthenticationFilter;
43  import org.springframework.web.filter.CorsFilter;
44  import org.zalando.problem.spring.web.advice.security.SecurityProblemSupport;
45  
46  import javax.annotation.PostConstruct;
47  
48  @Configuration
49  @Import(SecurityProblemSupport.class)
50  @EnableWebSecurity
51  @EnableGlobalMethodSecurity(prePostEnabled = true, securedEnabled = true)
52  public class SecurityConfiguration extends WebSecurityConfigurerAdapter {
53  
54      private final AuthenticationManagerBuilder authenticationManagerBuilder;
55  
56      private final UserDetailsService userDetailsService;
57  
58      private final TokenProvider tokenProvider;
59  
60      private final CorsFilter corsFilter;
61  
62      private final SecurityProblemSupport problemSupport;
63  
64      public SecurityConfiguration(AuthenticationManagerBuilder authenticationManagerBuilder, UserDetailsService userDetailsService,TokenProvider tokenProvider,CorsFilter corsFilter, SecurityProblemSupport problemSupport) {
65          this.authenticationManagerBuilder = authenticationManagerBuilder;
66          this.userDetailsService = userDetailsService;
67          this.tokenProvider = tokenProvider;
68          this.corsFilter = corsFilter;
69          this.problemSupport = problemSupport;
70      }
71  
72      @PostConstruct
73      public void init() {
74          try {
75              authenticationManagerBuilder
76                  .userDetailsService(userDetailsService)
77                  .passwordEncoder(passwordEncoder());
78          } catch (Exception e) {
79              throw new BeanInitializationException("Security configuration failed", e);
80          }
81      }
82  
83      @Bean
84      public PasswordEncoder passwordEncoder() {
85          return new BCryptPasswordEncoder();
86      }
87  
88      @Override
89      public void configure(WebSecurity web) throws Exception {
90          web.ignoring()
91              .antMatchers(HttpMethod.OPTIONS, "/**")
92              .antMatchers("/app/**/*.{js,html}")
93              .antMatchers("/i18n/**")
94              .antMatchers("/content/**")
95              .antMatchers("/swagger-ui/index.html")
96              .antMatchers("/test/**")
97              .antMatchers("/h2-console/**");
98      }
99  
100     @Override
101     protected void configure(HttpSecurity http) throws Exception {
102         http
103             .addFilterBefore(corsFilter, UsernamePasswordAuthenticationFilter.class)
104             .exceptionHandling()
105             .authenticationEntryPoint(problemSupport)
106             .accessDeniedHandler(problemSupport)
107         .and()
108             .csrf()
109             .disable()
110             .headers()
111             .frameOptions()
112             .disable()
113         .and()
114             .sessionManagement()
115             .sessionCreationPolicy(SessionCreationPolicy.STATELESS)
116         .and()
117             .authorizeRequests()
118             .antMatchers("/api/register").permitAll()
119             .antMatchers("/api/activate").permitAll()
120             .antMatchers("/api/authenticate").permitAll()
121             .antMatchers("/api/account/reset-password/init").permitAll()
122             .antMatchers("/api/account/reset-password/finish").permitAll()
123             .antMatchers("/api/profile-info").permitAll()
124             .antMatchers("/api/**").authenticated()
125             .antMatchers("/management/health").permitAll()
126             .antMatchers("/management/**").hasAuthority(AuthoritiesConstants.ADMIN)
127             .antMatchers("/v2/api-docs/**").permitAll()
128             .antMatchers("/swagger-resources/configuration/ui").permitAll()
129             .antMatchers("/swagger-ui/index.html").hasAuthority(AuthoritiesConstants.ADMIN)
130         .and()
131             .apply(securityConfigurerAdapter());
132 
133     }
134 
135     private JWTConfigurer securityConfigurerAdapter() {
136         return new JWTConfigurer(tokenProvider);
137     }
138 
139     @Bean
140     public SecurityEvaluationContextExtension securityEvaluationContextExtension() {
141         return new SecurityEvaluationContextExtension();
142     }
143 }
