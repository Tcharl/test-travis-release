1   package io.github.jhipster.sample.config;
2   
3   import io.github.jhipster.config.JHipsterProperties;
4   
5   import com.codahale.metrics.JmxReporter;
6   import com.codahale.metrics.JvmAttributeGaugeSet;
7   import com.codahale.metrics.MetricRegistry;
8   import com.codahale.metrics.Slf4jReporter;
9   import com.codahale.metrics.health.HealthCheckRegistry;
10  import com.codahale.metrics.jcache.JCacheGaugeSet;
11  import com.codahale.metrics.jvm.*;
12  import com.ryantenney.metrics.spring.config.annotation.EnableMetrics;
13  import com.ryantenney.metrics.spring.config.annotation.MetricsConfigurerAdapter;
14  import com.zaxxer.hikari.HikariDataSource;
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  import org.slf4j.Marker;
18  import org.slf4j.MarkerFactory;
19  import org.springframework.beans.factory.annotation.Autowired;
20  import org.springframework.context.annotation.*;
21  
22  import javax.annotation.PostConstruct;
23  import java.lang.management.ManagementFactory;
24  import java.util.concurrent.TimeUnit;
25  
26  @Configuration
27  @EnableMetrics(proxyTargetClass = true)
28  public class MetricsConfiguration extends MetricsConfigurerAdapter {
29  
30      private static final String PROP_METRIC_REG_JVM_MEMORY = "jvm.memory";
31      private static final String PROP_METRIC_REG_JVM_GARBAGE = "jvm.garbage";
32      private static final String PROP_METRIC_REG_JVM_THREADS = "jvm.threads";
33      private static final String PROP_METRIC_REG_JVM_FILES = "jvm.files";
34      private static final String PROP_METRIC_REG_JVM_BUFFERS = "jvm.buffers";
35      private static final String PROP_METRIC_REG_JVM_ATTRIBUTE_SET = "jvm.attributes";
36  
37      private static final String PROP_METRIC_REG_JCACHE_STATISTICS = "jcache.statistics";
38  
39      private final Logger log = LoggerFactory.getLogger(MetricsConfiguration.class);
40  
41      private MetricRegistry metricRegistry = new MetricRegistry();
42  
43      private HealthCheckRegistry healthCheckRegistry = new HealthCheckRegistry();
44  
45      private final JHipsterProperties jHipsterProperties;
46  
47      private HikariDataSource hikariDataSource;
48  
49      public MetricsConfiguration(JHipsterProperties jHipsterProperties) {
50          this.jHipsterProperties = jHipsterProperties;
51      }
52  
53      @Autowired(required = false)
54      public void setHikariDataSource(HikariDataSource hikariDataSource) {
55          this.hikariDataSource = hikariDataSource;
56      }
57  
58      @Override
59      @Bean
60      public MetricRegistry getMetricRegistry() {
61          return metricRegistry;
62      }
63  
64      @Override
65      @Bean
66      public HealthCheckRegistry getHealthCheckRegistry() {
67          return healthCheckRegistry;
68      }
69  
70      @PostConstruct
71      public void init() {
72          log.debug("Registering JVM gauges");
73          metricRegistry.register(PROP_METRIC_REG_JVM_MEMORY, new MemoryUsageGaugeSet());
74          metricRegistry.register(PROP_METRIC_REG_JVM_GARBAGE, new GarbageCollectorMetricSet());
75          metricRegistry.register(PROP_METRIC_REG_JVM_THREADS, new ThreadStatesGaugeSet());
76          metricRegistry.register(PROP_METRIC_REG_JVM_FILES, new FileDescriptorRatioGauge());
77          metricRegistry.register(PROP_METRIC_REG_JVM_BUFFERS, new BufferPoolMetricSet(ManagementFactory.getPlatformMBeanServer()));
78          metricRegistry.register(PROP_METRIC_REG_JVM_ATTRIBUTE_SET, new JvmAttributeGaugeSet());
79          metricRegistry.register(PROP_METRIC_REG_JCACHE_STATISTICS, new JCacheGaugeSet());
80          if (hikariDataSource != null) {
81              log.debug("Monitoring the datasource");
82              hikariDataSource.setMetricRegistry(metricRegistry);
83          }
84          if (jHipsterProperties.getMetrics().getJmx().isEnabled()) {
85              log.debug("Initializing Metrics JMX reporting");
86              JmxReporter jmxReporter = JmxReporter.forRegistry(metricRegistry).build();
87              jmxReporter.start();
88          }
89          if (jHipsterProperties.getMetrics().getLogs().isEnabled()) {
90              log.info("Initializing Metrics Log reporting");
91              Marker metricsMarker = MarkerFactory.getMarker("metrics");
92              final Slf4jReporter reporter = Slf4jReporter.forRegistry(metricRegistry)
93                  .outputTo(LoggerFactory.getLogger("metrics"))
94                  .markWith(metricsMarker)
95                  .convertRatesTo(TimeUnit.SECONDS)
96                  .convertDurationsTo(TimeUnit.MILLISECONDS)
97                  .build();
98              reporter.start(jHipsterProperties.getMetrics().getLogs().getReportFrequency(), TimeUnit.SECONDS);
99          }
100     }
101 }
