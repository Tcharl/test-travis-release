17  
18  @Configuration
19  @EnableCaching
20  @AutoConfigureAfter(value = { MetricsConfiguration.class })
21  @AutoConfigureBefore(value = { WebConfigurer.class, DatabaseConfiguration.class })
22  public class CacheConfiguration {
23  
24      private final javax.cache.configuration.Configuration<Object, Object> jcacheConfiguration;
25  
26      public CacheConfiguration(JHipsterProperties jHipsterProperties) {
27          JHipsterProperties.Cache.Ehcache ehcache =
28              jHipsterProperties.getCache().getEhcache();
29  
30          jcacheConfiguration = Eh107Configuration.fromEhcacheCacheConfiguration(
31              CacheConfigurationBuilder.newCacheConfigurationBuilder(Object.class, Object.class,
32                  ResourcePoolsBuilder.heap(ehcache.getMaxEntries()))
33                  .withExpiry(Expirations.timeToLiveExpiration(Duration.of(ehcache.getTimeToLiveSeconds(), TimeUnit.SECONDS)))
34                  .build());
35      }
36
