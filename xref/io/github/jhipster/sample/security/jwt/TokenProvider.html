1   package io.github.jhipster.sample.security.jwt;
2   
3   import io.github.jhipster.config.JHipsterProperties;
4   
5   import java.util.*;
6   import java.util.stream.Collectors;
7   import javax.annotation.PostConstruct;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
12  import org.springframework.security.core.Authentication;
13  import org.springframework.security.core.GrantedAuthority;
14  import org.springframework.security.core.authority.SimpleGrantedAuthority;
15  import org.springframework.security.core.userdetails.User;
16  import org.springframework.stereotype.Component;
17  
18  import io.jsonwebtoken.*;
19  
20  @Component
21  public class TokenProvider {
22  
23      private final Logger log = LoggerFactory.getLogger(TokenProvider.class);
24  
25      private static final String AUTHORITIES_KEY = "auth";
26  
27      private String secretKey;
28  
29      private long tokenValidityInMilliseconds;
30  
31      private long tokenValidityInMillisecondsForRememberMe;
32  
33      private final JHipsterProperties jHipsterProperties;
34  
35      public TokenProvider(JHipsterProperties jHipsterProperties) {
36          this.jHipsterProperties = jHipsterProperties;
37      }
38  
39      @PostConstruct
40      public void init() {
41          this.secretKey =
42              jHipsterProperties.getSecurity().getAuthentication().getJwt().getSecret();
43  
44          this.tokenValidityInMilliseconds =
45              1000 * jHipsterProperties.getSecurity().getAuthentication().getJwt().getTokenValidityInSeconds();
46          this.tokenValidityInMillisecondsForRememberMe =
47              1000 * jHipsterProperties.getSecurity().getAuthentication().getJwt().getTokenValidityInSecondsForRememberMe();
48      }
49  
50      public String createToken(Authentication authentication, Boolean rememberMe) {
51          String authorities = authentication.getAuthorities().stream()
52              .map(GrantedAuthority::getAuthority)
53              .collect(Collectors.joining(","));
54  
55          long now = (new Date()).getTime();
56          Date validity;
57          if (rememberMe) {
58              validity = new Date(now + this.tokenValidityInMillisecondsForRememberMe);
59          } else {
60              validity = new Date(now + this.tokenValidityInMilliseconds);
61          }
62  
63          return Jwts.builder()
64              .setSubject(authentication.getName())
65              .claim(AUTHORITIES_KEY, authorities)
66              .signWith(SignatureAlgorithm.HS512, secretKey)
67              .setExpiration(validity)
68              .compact();
69      }
70  
71      public Authentication getAuthentication(String token) {
72          Claims claims = Jwts.parser()
73              .setSigningKey(secretKey)
74              .parseClaimsJws(token)
75              .getBody();
76  
77          Collection<? extends GrantedAuthority> authorities =
78              Arrays.stream(claims.get(AUTHORITIES_KEY).toString().split(","))
79                  .map(SimpleGrantedAuthority::new)
80                  .collect(Collectors.toList());
81  
82          User principal = new User(claims.getSubject(), "", authorities);
83  
84          return new UsernamePasswordAuthenticationToken(principal, token, authorities);
85      }
86  
87      public boolean validateToken(String authToken) {
88          try {
89              Jwts.parser().setSigningKey(secretKey).parseClaimsJws(authToken);
90              return true;
91          } catch (SignatureException e) {
92              log.info("Invalid JWT signature.");
93              log.trace("Invalid JWT signature trace: {}", e);
94          } catch (MalformedJwtException e) {
95              log.info("Invalid JWT token.");
96              log.trace("Invalid JWT token trace: {}", e);
97          } catch (ExpiredJwtException e) {
98              log.info("Expired JWT token.");
99              log.trace("Expired JWT token trace: {}", e);
100         } catch (UnsupportedJwtException e) {
101             log.info("Unsupported JWT token.");
102             log.trace("Unsupported JWT token trace: {}", e);
103         } catch (IllegalArgumentException e) {
104             log.info("JWT token compact of handler are invalid.");
105             log.trace("JWT token compact of handler are invalid trace: {}", e);
106         }
107         return false;
108     }
109 }
