181     @Test
182     @Transactional
183     public void createBankAccount() throws Exception {
184         int databaseSizeBeforeCreate = bankAccountRepository.findAll().size();
185 
186         // Create the BankAccount
187         BankAccountDTO bankAccountDTO = bankAccountMapper.toDto(bankAccount);
188         restBankAccountMockMvc.perform(post("/api/bank-accounts")
189             .contentType(TestUtil.APPLICATION_JSON_UTF8)
190             .content(TestUtil.convertObjectToJsonBytes(bankAccountDTO)))
191             .andExpect(status().isCreated());
192 
193         // Validate the BankAccount in the database
194         List<BankAccount> bankAccountList = bankAccountRepository.findAll();
195         assertThat(bankAccountList).hasSize(databaseSizeBeforeCreate + 1);
196         BankAccount testBankAccount = bankAccountList.get(bankAccountList.size() - 1);
197         assertThat(testBankAccount.getName()).isEqualTo(DEFAULT_NAME);
198         assertThat(testBankAccount.getBankNumber()).isEqualTo(DEFAULT_BANK_NUMBER);
199         assertThat(testBankAccount.getAgencyNumber()).isEqualTo(DEFAULT_AGENCY_NUMBER);
200         assertThat(testBankAccount.getLastOperationDuration()).isEqualTo(DEFAULT_LAST_OPERATION_DURATION);
201         assertThat(testBankAccount.getMeanOperationDuration()).isEqualTo(DEFAULT_MEAN_OPERATION_DURATION);
202         assertThat(testBankAccount.getBalance()).isEqualTo(DEFAULT_BALANCE);
203         assertThat(testBankAccount.getOpeningDay()).isEqualTo(DEFAULT_OPENING_DAY);
204         assertThat(testBankAccount.getLastOperationDate()).isEqualTo(DEFAULT_LAST_OPERATION_DATE);
205         assertThat(testBankAccount.isActive()).isEqualTo(DEFAULT_ACTIVE);
206         assertThat(testBankAccount.getAccountType()).isEqualTo(DEFAULT_ACCOUNT_TYPE);
207         assertThat(testBankAccount.getAttachment()).isEqualTo(DEFAULT_ATTACHMENT);
208         assertThat(testBankAccount.getAttachmentContentType()).isEqualTo(DEFAULT_ATTACHMENT_CONTENT_TYPE);
209         assertThat(testBankAccount.getDescription()).isEqualTo(DEFAULT_DESCRIPTION);
210     }
