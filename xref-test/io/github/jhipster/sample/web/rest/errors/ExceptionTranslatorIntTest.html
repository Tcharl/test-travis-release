25   * @see ExceptionTranslator

27  @RunWith(SpringRunner.class)
28  @SpringBootTest(classes = JhipsterSampleApplicationApp.class)
29  public class ExceptionTranslatorIntTest {
30  
31      @Autowired
32      private ExceptionTranslatorTestController controller;
33  
34      @Autowired
35      private ExceptionTranslator exceptionTranslator;
36  
37      @Autowired
38      private MappingJackson2HttpMessageConverter jacksonMessageConverter;
39  
40      private MockMvc mockMvc;
41  
42      @Before
43      public void setup() {
44          mockMvc = MockMvcBuilders.standaloneSetup(controller)
45              .setControllerAdvice(exceptionTranslator)
46              .setMessageConverters(jacksonMessageConverter)
47              .build();
48      }
49  
50      @Test
51      public void testConcurrencyFailure() throws Exception {
52          mockMvc.perform(get("/test/concurrency-failure"))
53              .andExpect(status().isConflict())
54              .andExpect(content().contentType(MediaTypes.PROBLEM))
55              .andExpect(jsonPath("$.message").value(ErrorConstants.ERR_CONCURRENCY_FAILURE));
56      }
57  
58      @Test
59      public void testMethodArgumentNotValid() throws Exception {
60           mockMvc.perform(post("/test/method-argument").content("{}").contentType(MediaType.APPLICATION_JSON))
61               .andExpect(status().isBadRequest())
62               .andExpect(content().contentType(MediaTypes.PROBLEM))
63               .andExpect(jsonPath("$.message").value(ErrorConstants.ERR_VALIDATION))
64               .andExpect(jsonPath("$.fieldErrors.[0].objectName").value("testDTO"))
65               .andExpect(jsonPath("$.fieldErrors.[0].field").value("test"))
66               .andExpect(jsonPath("$.fieldErrors.[0].message").value("NotNull"));
67      }
68  
69      @Test
70      public void testParameterizedError() throws Exception {
71          mockMvc.perform(get("/test/parameterized-error"))
72              .andExpect(status().isBadRequest())
73              .andExpect(content().contentType(MediaTypes.PROBLEM))
74              .andExpect(jsonPath("$.message").value("test parameterized error"))
75              .andExpect(jsonPath("$.params.param0").value("param0_value"))
76              .andExpect(jsonPath("$.params.param1").value("param1_value"));
77      }
